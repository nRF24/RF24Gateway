175  return fd;

179

181

183  strncpy(ifr.ifr_name, dev, IFNAMSIZ);
184  }
185 
186  // Create device

194  }
195 
196  //Make persistent
197  if(ioctl(fd, TUNSETPERSIST, 1) < 0){
198  #if (DEBUG >= 1)
199  std::cerr << "RF24Gw: Error: enabling TUNSETPERSIST" << std::endl;
200  #endif
201  return -1;
202  }
203 
204  if(!config_TUN){
205  struct sockaddr sap;

207  ((char*)sap.sa_data)[4]=address;

219  fprintf(stderr, "RF24Gw: Failed to set MAC address\n");

230 int RF24Gateway::setIP( char *ip_addr, char *mask) {
231 
232  struct ifreq ifr;
233  struct sockaddr_in sin;
234  int sockfd = socket(AF_INET, SOCK_DGRAM, 0);
235  if(sockfd == -1){
236  fprintf(stderr, "Could not get socket.\n");
237  return -1;
238  }
239 
240 
241  sin.sin_family = AF_INET;
242  //inet_aton(ip_addr,&sin.sin_addr.s_addr);
243  inet_aton(ip_addr,&sin.sin_addr);
244  strncpy(ifr.ifr_name, tunName, IFNAMSIZ);
245 
246  if (ioctl(sockfd, SIOCGIFFLAGS, &ifr) < 0) {
247  fprintf(stderr, "ifdown: shutdown ");
248  perror(ifr.ifr_name);
249  return -1;
250  }
251 
252  #ifdef ifr_flags
253  # define IRFFLAGS ifr_flags
254  #else /* Present on kFreeBSD */
255  # define IRFFLAGS ifr_flagshigh
256  #endif
257 
258  if (!(ifr.IRFFLAGS & IFF_UP)) {
259  //fprintf(stdout, "Device is currently down..setting up.-- %u\n",ifr.IRFFLAGS);
260  ifr.IRFFLAGS |= IFF_UP;
261  if (ioctl(sockfd, SIOCSIFFLAGS, &ifr) < 0) {
262  fprintf(stderr, "ifup: failed ");
263  perror(ifr.ifr_name);
264  return -1;
265  }
266  }
267 
268  memcpy(&ifr.ifr_addr, &sin, sizeof(struct sockaddr));
269 
270  // Set interface address
271  if (ioctl(sockfd, SIOCSIFADDR, &ifr) < 0) {
272  fprintf(stderr, "Cannot set IP address. ");
273  perror(ifr.ifr_name);
274  return -1;
275  }
276 
277  inet_aton(mask, &sin.sin_addr);
278  memcpy(&ifr.ifr_addr, &sin, sizeof(struct sockaddr));
279 
280  if (ioctl(sockfd, SIOCSIFNETMASK, &ifr) < 0)
281  {
282  fprintf(stderr,"Cannot define subnet mask for this device");
283  perror(ifr.ifr_name);
284  return -1;
285  }
286 
287  #undef IRFFLAGS
288  return 0;
289 }

434 void RF24Gateway::handleRadioOut(){
435 
436  bool ok = 0;
437 
438  while(!txQueue.empty() && network.external_queue.size() == 0) {
439 
440 
441  msgStruct *msgTx = &txQueue.front();
442 
443  #if (DEBUG >= 1)
444  std::cout << "Radio: Sending " << msgTx->size << " bytes ... ";
445  std::cout << std::endl;
446  #endif
447  #if (DEBUG >= 3)
448  //PrintDebug == 1 does not have an endline.
449  //printPayload(msg.getPayloadStr(),"radio TX");
450  #endif
451 
452  std::uint8_t *tmp = msgTx->message;
453 
454 
455  if(!config_TUN ){ //TAP can use RF24Mesh for address assignment, but will still use ARP for address resolution
456 
457  uint32_t RF24_STR = 0x34324652; //Identifies the mac as an RF24 mac
458  uint32_t ARP_BC = 0xFFFFFFFF; //Broadcast address
459  struct macStruct{
460  uint32_t rf24_Verification;
461  uint16_t rf24_Addr;
462  };
463 
464 
465  macStruct macData;
466  memcpy(&macData.rf24_Addr,tmp+4,2);
467  memcpy(&macData.rf24_Verification,tmp,4);
468 
469 
470  if(macData.rf24_Verification == RF24_STR){
471  const uint16_t other_node = macData.rf24_Addr;
472  RF24NetworkHeader header(/*to node*/ other_node, EXTERNAL_DATA_TYPE);
473  ok = network.write(header,&msgTx->message,msgTx->size);
474 
475  }else
476  if(macData.rf24_Verification == ARP_BC){
477  RF24NetworkHeader header(/*to node*/ 00, EXTERNAL_DATA_TYPE); //Set to master node, will be modified by RF24Network if multi-casting
478  if(msgTx->size <= 42){
479  if(thisNodeAddress == 00){ //Master Node
480 
481  uint32_t arp_timeout = millis();
482 
483  ok=network.multicast(header,&msgTx->message,msgTx->size,1 ); //Send to Level 1
484  while(millis() - arp_timeout < 5){network.update();}
485  network.multicast(header,&msgTx->message,msgTx->size,1 ); //Send to Level 1
486  arp_timeout=millis();
487  while(millis()- arp_timeout < 15){network.update();}
488  network.multicast(header,&msgTx->message,msgTx->size,1 ); //Send to Level 1
489 
490  }else{
491 
492  ok = network.write(header,&msgTx->message,msgTx->size);
493  }
494  }
495  }
496  }else{ // TUN always needs to use RF24Mesh for address assignment AND resolution
497 
498  uint8_t lastOctet = tmp[19];
499  int16_t meshAddr;
500 
501  RF24NetworkHeader header(00, EXTERNAL_DATA_TYPE);
502  bool sendData = false;
503 
504  struct in_addr ipDestination;
505  memcpy(&ipDestination.s_addr,&tmp[16],4);
506 
507  if( (getLocalIP().s_addr & 0x00FFFFFF) == (ipDestination.s_addr & 0x00FFFFFF)){ //Is inside the RF24Mesh network
508  if ( (meshAddr = mesh.getAddress(lastOctet)) > 0 ) {
509  header.to_node = meshAddr;
510  sendData = true;
511  }else{
512  if(thisNodeID > 0){//If IP is in mesh range, address lookup fails, and this is not master,
513  sendData = true; //send to 00 anyway in case destination is master, or the lookup just failed
514  }
515  //printf("Could not find matching mesh nodeID for IP ending in %d\n",lastOctet);
516  }
517  }else
518  if( thisNodeID > 0 ){ //If not master, send to master for routing etc. if target not within mesh
519  sendData = true;
520  }else
521  if(routingTableSize > 0){
522  for(int i=0; i<routingTableSize; i++){
523  struct in_addr network;
524  network.s_addr = routingStruct[i].ip.s_addr & routingStruct[i].mask.s_addr;
525  struct in_addr destNet;
526  destNet.s_addr = ipDestination.s_addr & routingStruct[i].mask.s_addr;
527  //printf("network %s destNet: %s\n",inet_ntoa(network),inet_ntoa(destNet));
528  if( destNet.s_addr == network.s_addr){
529  uint8_t toNode = routingStruct[i].gw.s_addr >> 24;
530  int16_t netAddr = 0;
531  if( (netAddr = mesh.getAddress(toNode)) > 0 ){
532  header.to_node = netAddr;
533  sendData = true;
534  break;
535  }
536  }
537  }
538  }
539 
540  if(sendData){
541  ok = network.write(header, msgTx->message, msgTx->size);
542  //std::cout << "SendData " << header.to_node << std::endl;
543  }
544 
545 
546  }
547  //delay( rf24_min(msgTx->size/48,20));
548  txQueue.pop();
549 
550 
551  //printf("Addr: 0%#x\n",macData.rf24_Addr);
552  //printf("Verif: 0%#x\n",macData.rf24_Verification);
553  if (ok) {
554  // std::cout << "ok." << std::endl;
555  } else {
556  // std::cerr << "failed." << std::endl;
557  }
558 
559 
560 
561  } //End Tx
