182 int RF24Gateway::allocateTunDevice(char *dev, int flags, uint16_t address)
183 {
184  struct ifreq ifr;
185  int fd;
186 
187  //open the device
188  if ((fd = open("/dev/net/tun", O_RDWR)) < 0) {
189  return fd;
190  }
191 
192  memset(&ifr, 0, sizeof(ifr));
193 
194  ifr.ifr_flags = flags; // IFF_TUN or IFF_TAP, plus maybe IFF_NO_PI
195 
196  if (*dev) {
197  strncpy(ifr.ifr_name, dev, IFNAMSIZ);
198  }
199 
200  // Create device
201  if (ioctl(fd, TUNSETIFF, (void *)&ifr) < 0) {
202  //close(fd);
203  //#if (DEBUG >= 1)
204  std::cerr << "RF24Gw: Error: enabling TUNSETIFF" << std::endl;
205  std::cerr << "RF24Gw: If changing from TAP/TUN, run 'sudo ip link delete tun_nrf24' to remove the interface" << std::endl;
206  return -1;
207  //#endif
208  }
209 
210  //Make persistent
211  if (ioctl(fd, TUNSETPERSIST, 1) < 0) {
212  #if (DEBUG >= 1)
213  std::cerr << "RF24Gw: Error: enabling TUNSETPERSIST" << std::endl;
214  #endif
215  return -1;
216  }
217 
218  if (!config_TUN) {
219  struct sockaddr sap;
220  sap.sa_family = ARPHRD_ETHER;
221  ((char *)sap.sa_data)[4] = address;
222  ((char *)sap.sa_data)[5] = address >> 8;
223  ((char *)sap.sa_data)[0] = 0x52;
224  ((char *)sap.sa_data)[1] = 0x46;
225  ((char *)sap.sa_data)[2] = 0x32;
226  ((char *)sap.sa_data)[3] = 0x34;
227 
228  //printf("Address 0%o first %u last %u\n",address,sap.sa_data[0],sap.sa_data[1]);
229  memcpy((char *)&ifr.ifr_hwaddr, (char *)&sap, sizeof(struct sockaddr));
230 
231  if (ioctl(fd, SIOCSIFHWADDR, &ifr) < 0) {
232  #if DEBUG >= 1
233  fprintf(stderr, "RF24Gw: Failed to set MAC address\n");
234  #endif
235  }
236  }
237 
238  strcpy(dev, ifr.ifr_name);
239  return fd;
240 }

244 int RF24Gateway::setIP(char *ip_addr, char *mask)
245 {
246  struct ifreq ifr;
247  struct sockaddr_in sin;
248  int sockfd = socket(AF_INET, SOCK_DGRAM, 0);
249  if (sockfd == -1) {
250  fprintf(stderr, "Could not get socket.\n");
251  return -1;
252  }
253 
254  sin.sin_family = AF_INET;
255  //inet_aton(ip_addr,&sin.sin_addr.s_addr);
256  inet_aton(ip_addr, &sin.sin_addr);
257  strncpy(ifr.ifr_name, tunName, IFNAMSIZ);
258 
259  if (ioctl(sockfd, SIOCGIFFLAGS, &ifr) < 0) {
260  fprintf(stderr, "ifdown: shutdown ");
261  perror(ifr.ifr_name);
262  return -1;
263  }
264 
265 #ifdef ifr_flags
266 #define IRFFLAGS ifr_flags
267 #else /* Present on kFreeBSD */
268 #define IRFFLAGS ifr_flagshigh
269 #endif
270 
271  if (!(ifr.IRFFLAGS & IFF_UP)) {
272  //fprintf(stdout, "Device is currently down..setting up.-- %u\n",ifr.IRFFLAGS);
273  ifr.IRFFLAGS |= IFF_UP;
274  if (ioctl(sockfd, SIOCSIFFLAGS, &ifr) < 0) {
275  fprintf(stderr, "ifup: failed ");
276  perror(ifr.ifr_name);
277  return -1;
278  }
279  }
280 
281  memcpy(&ifr.ifr_addr, &sin, sizeof(struct sockaddr));
282 
283  // Set interface address
284  if (ioctl(sockfd, SIOCSIFADDR, &ifr) < 0) {
285  fprintf(stderr, "Cannot set IP address. ");
286  perror(ifr.ifr_name);
287  return -1;
288  }
289 
290  inet_aton(mask, &sin.sin_addr);
291  memcpy(&ifr.ifr_addr, &sin, sizeof(struct sockaddr));
292 
293  if (ioctl(sockfd, SIOCSIFNETMASK, &ifr) < 0) {
294  fprintf(stderr, "Cannot define subnet mask for this device");
295  perror(ifr.ifr_name);
296  return -1;
297  }
298 
299 #undef IRFFLAGS
300  return 0;
301 }

424 struct in_addr RF24Gateway::getLocalIP()
425 {
426  struct ifaddrs *ifap, *ifa;
427  int family, s, n;
428  char host[NI_MAXHOST];
429  struct in_addr myNet;
430 
431  getifaddrs(&ifap);
432  for (ifa = ifap, n = 0; ifa != NULL; ifa = ifa->ifa_next, n++) {
433  if (std::string("tun_nrf24").compare(ifa->ifa_name) != 0 || ifa->ifa_addr == NULL) {
434  if (ifa->ifa_next == NULL) {
435  break;
436  }
437  else {
438  continue;
439  }
440  }
441 
442  family = ifa->ifa_addr->sa_family;
443 
444  //This is an IPv4 interface, get the IP
445  if (family == AF_INET) {
446  s = getnameinfo(ifa->ifa_addr, sizeof(struct sockaddr_in), host, NI_MAXHOST, NULL, 0, NI_NUMERICHOST);
447  if (s == 0) {
448  myNet.s_addr = ntohl(inet_network(host));
449  freeifaddrs(ifap);
450  return myNet;
451  }
452  }
453  }
454  freeifaddrs(ifap);
455  return myNet;

460 void RF24Gateway::handleRadioOut()
461 {
462  bool ok = 0;
463 
464  while (!txQueue.empty() && network.external_queue.size() == 0) {
465 
466  msgStruct *msgTx = &txQueue.front();
467 
468  #if (DEBUG >= 1)
469  std::cout << "Radio: Sending " << msgTx->size << " bytes ... ";
470  std::cout << std::endl;
471  #endif
472  #if (DEBUG >= 3)
473  //PrintDebug == 1 does not have an endline.
474  //printPayload(msg.getPayloadStr(),"radio TX");
475  #endif
476 
477  std::uint8_t *tmp = msgTx->message;
478 
479  if (!config_TUN) { //TAP can use RF24Mesh for address assignment, but will still use ARP for address resolution
480 
481  uint32_t RF24_STR = 0x34324652; //Identifies the mac as an RF24 mac
482  uint32_t ARP_BC = 0xFFFFFFFF; //Broadcast address
483  struct macStruct {
484  uint32_t rf24_Verification;
485  uint16_t rf24_Addr;
486  };
487 
488  macStruct macData;
489  memcpy(&macData.rf24_Addr, tmp + 4, 2);
490  memcpy(&macData.rf24_Verification, tmp, 4);
491 
492  if (macData.rf24_Verification == RF24_STR) {
493  const uint16_t other_node = macData.rf24_Addr;
494  RF24NetworkHeader header(/*to node*/ other_node, EXTERNAL_DATA_TYPE);
495  ok = network.write(header, &msgTx->message, msgTx->size);
496  }
497  else if (macData.rf24_Verification == ARP_BC) {
498  RF24NetworkHeader header(/*to node*/ 00, EXTERNAL_DATA_TYPE); //Set to master node, will be modified by RF24Network if multi-casting
499  if (msgTx->size <= 42) {
500  if (thisNodeAddress == 00) { //Master Node
501 
502  uint32_t arp_timeout = millis();
503 
504  ok = network.multicast(header, &msgTx->message, msgTx->size, 1); //Send to Level 1
505  while (millis() - arp_timeout < 5) {
506  network.update();
507  }
508  network.multicast(header, &msgTx->message, msgTx->size, 1); //Send to Level 1
509  arp_timeout = millis();
510  while (millis() - arp_timeout < 15) {
511  network.update();
512  }
513  network.multicast(header, &msgTx->message, msgTx->size, 1); //Send to Level 1
514  }
515  else {
516  ok = network.write(header, &msgTx->message, msgTx->size);
517  }
518  }
519  }
520  }
521  else { // TUN always needs to use RF24Mesh for address assignment AND resolution
522 
523  uint8_t lastOctet = tmp[19];
524  int16_t meshAddr;
525 
526  RF24NetworkHeader header(00, EXTERNAL_DATA_TYPE);
527  bool sendData = false;
528 
529  struct in_addr ipDestination;
530  memcpy(&ipDestination.s_addr, &tmp[16], 4);
531 
532  if ((getLocalIP().s_addr & 0x00FFFFFF) == (ipDestination.s_addr & 0x00FFFFFF)) { //Is inside the RF24Mesh network
533  if ((meshAddr = mesh.getAddress(lastOctet)) > 0) {
534  header.to_node = meshAddr;
535  sendData = true;
536  }
537  else {
538  if (thisNodeID > 0) { //If IP is in mesh range, address lookup fails, and this is not master,
539  sendData = true; //send to 00 anyway in case destination is master, or the lookup just failed
540  }
541  //printf("Could not find matching mesh nodeID for IP ending in %d\n",lastOctet);
542  }
543  }
544  else if (thisNodeID > 0) { //If not master, send to master for routing etc. if target not within mesh
545  sendData = true;
546  }
547  else if (routingTableSize > 0) {
548  for (int i = 0; i < routingTableSize; i++) {
549  struct in_addr network;
550  network.s_addr = routingStruct[i].ip.s_addr & routingStruct[i].mask.s_addr;
551  struct in_addr destNet;
552  destNet.s_addr = ipDestination.s_addr & routingStruct[i].mask.s_addr;
553  //printf("network %s destNet: %s\n",inet_ntoa(network),inet_ntoa(destNet));
554  if (destNet.s_addr == network.s_addr) {
555  uint8_t toNode = routingStruct[i].gw.s_addr >> 24;
556  int16_t netAddr = 0;
557  if ((netAddr = mesh.getAddress(toNode)) > 0) {
558  header.to_node = netAddr;
559  sendData = true;
560  break;
561  }
562  }
563  }
564  }
565 
566  if (sendData) {
567  ok = network.write(header, msgTx->message, msgTx->size);
568  //std::cout << "SendData " << header.to_node << std::endl;
569  }
570  }
571  //delay( rf24_min(msgTx->size/48,20));
572  txQueue.pop();
573 
574  //printf("Addr: 0%#x\n",macData.rf24_Addr);
575  //printf("Verif: 0%#x\n",macData.rf24_Verification);
576  if (ok) {
577  // std::cout << "ok." << std::endl;
578  }
579  else {
580  // std::cerr << "failed." << std::endl;
581  }
582 
583  } //End Tx
584 }

588 void RF24Gateway::handleRX(uint32_t waitDelay)
589 {
590  fd_set socketSet;
591  struct timeval selectTimeout;
592  uint8_t buffer[MAX_PAYLOAD_SIZE];
593  int nread;
594 
595  FD_ZERO(&socketSet);
596  FD_SET(tunFd, &socketSet);
597 
598  selectTimeout.tv_sec = 0;
599  selectTimeout.tv_usec = waitDelay * 1000;
600 
601  if (select(tunFd + 1, &socketSet, NULL, NULL, &selectTimeout) != 0) {
602  if (FD_ISSET(tunFd, &socketSet)) {
603  if ((nread = read(tunFd, buffer, MAX_PAYLOAD_SIZE)) >= 0) {
604 
605  #if (DEBUG >= 1)
606  std::cout << "Tun: Successfully read " << nread << " bytes from tun device" << std::endl;
607  #endif
608  #if (DEBUG >= 3)
609  std::cout << "TunRead: " << std::endl;
610  for (int i = 0; i < nread; i++)
611  {
612  printf(":%0x :", buffer[i]);
613  }
614  std::cout << std::endl;
615  #endif
616  msgStruct msg;
617  memcpy(&msg.message, &buffer, nread);
618  msg.size = nread;
619  if (txQueue.size() < 2) {
620  txQueue.push(msg);
621  }
622  else {
623  droppedIncoming++;
624  }
625  }
626  else {
627  #if (DEBUG >= 1)
628  std::cerr << "Tun: Error while reading from tun/tap interface." << std::endl;
629  #endif
630  }
631  }
632  }
