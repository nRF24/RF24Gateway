157  /**
158  * If a user has created a file "routing.txt" in the RF24Gateway working directory, it will be loaded <br>
159  * at startup into the routing table. The file should contain standard routing table entries as follows: <br>
160  * IP<space>NetMask<space>Gateway <br>
161  * ie: routing.txt
162  * @code
163  * 10.1.4.0 255.255.255.0 10.1.3.33
164  * 0.0.0.0 0.0.0.0 10.1.3.34 @endcode
165  *
166  * The first example entry would route all traffic to the 10.1.4.x subnet to 10.1.3.33 <br>
167  * All other traffic not destined for the RF24 nodes will use 10.1.3.34 as the gateway
168  *
169  * Data can be accessed using standard linux Internet address manipulation routines as follows:
170  * @code
171  * printf("**IP\t\tMask\t\tGateway**\n");
172  * for(int i=0; i<gw.routingTableSize; i++){
173  * printf("%s \t",inet_ntoa(gw.routingStruct[i].ip));//inet_ntoa uses a statically assigned buffer, so the printf calls need to be done separately
174  * printf("%s \t",inet_ntoa(gw.routingStruct[i].mask));
175  * printf("%s\n", inet_ntoa(gw.routingStruct[i].gw));
176  * //std::cout << inet_ntoa(gw.routingStruct[i].ip) << " \t" << inet_ntoa(gw.routingStruct[i].mask) << " \t" << inet_ntoa(gw.routingStruct[i].gw) << std::endl;
177  * }
178  * printf("*****\n");
179  *
180  * Users can modify the routing table as desired, but changes made in code will not currently be saved to file
181  *
182  * @endcode
183  *
184  */

187  struct in_addr mask;
188  struct in_addr gw;
